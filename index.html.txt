function showToast(message, type) {
    var container = document.getElementById('toastContainer');
    var toast = document.createElement('div');
    toast.className = 'toast toast-' + type;
    var icon = type === 'success' ? 'check-circle' : type === 'error' ? 'exclamation-circle' : 'info-circle';
    toast.innerHTML = '<i class="fas fa-' + icon + '"></i> ' + message;
    container.appendChild(toast);
    setTimeout(function() { toast.remove(); }, 3000);
}

function formatDate(dateStr) {
    if (!dateStr) return '---';
    try {
        var d = new Date(dateStr);
        if (isNaN(d.getTime())) return dateStr;
        return d.toLocaleDateString('ar-SA', { year: 'numeric', month: '2-digit', day: '2-digit' });
    } catch(e) { return dateStr; }
}

function daysLeft(dateStr) {
    if (!dateStr) return null;
    var d = new Date(dateStr);
    var now = new Date();
    var diff = Math.ceil((d - now) / (1000 * 60 * 60 * 24));
    return diff;
}

window.handleSheetsData = function(data) {
    var btn = document.getElementById('refreshBtn');
    btn.classList.remove('btn-refresh-spin');

    if(data && data.success && data.posts) {
        allPosts = data.posts.map(normalizePost);
        renderDashboard();
        showToast('تم التحديث بنجاح! ' + allPosts.length + ' منشور', 'success');
    } else if (data && data.error) {
        showToast('خطأ: ' + data.error, 'error');
        showError('خطأ من Google Sheets: ' + data.error);
    } else {
        showToast('بيانات غير صالحة', 'error');
    }

    if(scriptTag && scriptTag.parentNode) {
        scriptTag.parentNode.removeChild(scriptTag);
        scriptTag = null;
    }
};

function loadData() {
    var btn = document.getElementById('refreshBtn');
    btn.classList.add('btn-refresh-spin');

    var board = document.getElementById('kanbanBoard');
    board.innerHTML = '<div class="loading-container"><div class="shimmer-wrapper"><div class="shimmer-circle"></div><div class="shimmer-line"></div><div class="shimmer-line" style="width:150px"></div></div></div>';

    if(scriptTag && scriptTag.parentNode) scriptTag.parentNode.removeChild(scriptTag);

    scriptTag = document.createElement('script');
    scriptTag.src = SHEETS_API_URL + '?callback=handleSheetsData&t=' + Date.now();
    scriptTag.onerror = function() {
        showToast('تعذر التحميل', 'error');
        showError('تعذر تحميل البيانات');
        btn.classList.remove('btn-refresh-spin');
    };

    document.head.appendChild(scriptTag);

    setTimeout(function() {
        if(btn.classList.contains('btn-refresh-spin')) {
            showToast('انتهى الوقت', 'error');
            btn.classList.remove('btn-refresh-spin');
        }
    }, 10000);
}

}

function normalizePost(row) {
    var getValue = function(keys) {
        for (var i = 0; i < keys.length; i++) {
            if (row[keys[i]] !== undefined && row[keys[i]] !== '') return row[keys[i]];
        }
        return '';
    };
    return {
        id: row['id'] || Date.now() + Math.random(),
        platform: getValue(['المنصة', 'platform']).toLowerCase().trim(),
        type: (getValue(['النوع', 'type']).toLowerCase().trim()) || 'post',
        title: getValue(['العنوان', 'title']) || 'بدون عنوان',
        desc: getValue(['الفكرة', 'desc', 'description']),
        status: statusMap[getValue(['الحالة', 'status']).trim()] || 'planned',
        date: getValue(['التاريخ', 'date']),
        designLink: getValue(['رابط التصميم', 'رابط_التصميم', 'designLink']),
        finalLink: getValue(['رابط النهائي', 'رابط_النهائي', 'finalLink']),
        designer: getValue(['المصمم', 'designer']),
        buyer: getValue(['الميديا باير', 'الميديا_باير', 'buyer'])
    };
}

function renderDashboard() {
    updateStats();
    updateProgress();
    renderKanban();

function updateStats() {
    var insta = allPosts.filter(function(p) { return p.platform.indexOf('insta') !== -1; }).length;
    var tiktok = allPosts.filter(function(p) { return p.platform.indexOf('tik') !== -1; }).length;
    var designing = allPosts.filter(function(p) { return p.status === 'designing'; }).length;
    var producing = allPosts.filter(function(p) { return p.status === 'producing'; }).length;
    var delayed = allPosts.filter(function(p) {
        if(!p.date || p.status === 'posted') return false;
        return new Date(p.date) < new Date();
    }).length;

    document.getElementById('statInsta').textContent = insta;
    document.getElementById('statTiktok').textContent = tiktok;
    document.getElementById('statDesign').textContent = designing;
    document.getElementById('statProd').textContent = producing;
    document.getElementById('statDelay').textContent = delayed;
}

function updateProgress() {
    var total = allPosts.length;
    if (total === 0) return;
    var posted = allPosts.filter(function(p) { return p.status === 'posted'; }).length;
    var percent = Math.round((posted / total) * 100);

    document.getElementById('progressSection').style.display = 'block';
    document.getElementById('progressValue').textContent = percent + '%';
    document.getElementById('progressBar').style.width = percent + '%';
}

function renderKanban() {
    var statuses = ['planned', 'designing', 'ready', 'producing', 'scheduled', 'posted'];
    var board = document.getElementById('kanbanBoard');

    var filtered = allPosts;
    if(currentFilter === 'instagram') filtered = allPosts.filter(function(p) { return p.platform.indexOf('insta') !== -1; });
    if(currentFilter === 'tiktok') filtered = allPosts.filter(function(p) { return p.platform.indexOf('tik') !== -1; });
    if(currentFilter === 'designer') filtered = allPosts.filter(function(p) { return p.status === 'planned' || p.status === 'designing'; });
    if(currentFilter === 'buyer') filtered = allPosts.filter(function(p) { return p.status === 'ready' || p.status === 'producing'; });

    board.innerHTML = statuses.map(function(status) {
        var statusPosts = filtered.filter(function(p) { return p.status === status; });
        return '<div class="kanban-column"><div class="kanban-header"><span>' + statusLabels[status] + '</span><span class="kanban-count">' + statusPosts.length + '</span></div>' + (statusPosts.length === 0 ? '<div class="empty-state"><i class="fas fa-inbox"></i><p>لا توجد منشورات</p></div>' : statusPosts.map(function(p, i) { return createCardHTML(p, i); }).join('')) + '</div>';
    }).join('');
}

function createCardHTML(p, index) {
    var isDelayed = p.date && new Date(p.date) < new Date() && p.status !== 'posted';
    var typeKey = typeLabels[p.type] ? p.type : 'post';
    var displayDate = formatDate(p.date);
    var days = daysLeft(p.date);
    var daysText = days !== null && days > 0 ? ' (تبقى ' + days + ' يوم)' : '';
    var dateClass = isDelayed ? 'urgent' : '';

    var html = '<div class="content-card ' + (isDelayed ? 'delayed' : '') + '" onclick="openModal(' + p.id + ')" style="animation-delay:' + (index * 0.05) + 's">';
    html += '<span class="card-type type-' + typeKey + '"><i class="fas fa-' + (typeIcons[typeKey] || 'file') + '"></i> ' + (typeLabels[typeKey] || p.type) + '</span>';
    html += '<div class="card-title">' + p.title + '</div>';
    html += '<div class="card-desc">' + (p.desc || 'لا يوجد وصف') + '</div>';
    html += '<div class="card-meta">';
    html += '<span class="card-platform"><i class="fab fa-' + (p.platform.indexOf('tik') !== -1 ? 'tiktok' : 'instagram') + '"></i> ' + p.platform + '</span>';
    html += '<span class="card-date ' + dateClass + '"><i class="fas fa-calendar"></i> ' + displayDate + daysText + '</span>';
    html += '</div>';
    if(p.designer || p.buyer) {
        html += '<div class="card-assignees">';
        if(p.designer) html += '<span class="assignee-tag"><i class="fas fa-paint-brush"></i>' + p.designer + '</span>';
        if(p.buyer) html += '<span class="assignee-tag"><i class="fas fa-ad"></i>' + p.buyer + '</span>';
        html += '</div>';
    }
    if(p.designLink || p.finalLink) {
        html += '<div class="card-links">';
        if(p.designLink) html += '<a href="' + p.designLink + '" target="_blank" class="link-btn" onclick="event.stopPropagation()"><i class="fas fa-paint-brush"></i> التصميم</a>';
        if(p.finalLink) html += '<a href="' + p.finalLink + '" target="_blank" class="link-btn" onclick="event.stopPropagation()"><i class="fas fa-video"></i> النهائي</a>';
        html += '</div>';
    }
    html += '</div>';
    return html;
}

function searchPosts() {
    var query = document.getElementById('searchInput').value.toLowerCase();
    if (!query) {
        renderKanban();
        return;
    }

    var statuses = ['planned', 'designing', 'ready', 'producing', 'scheduled', 'posted'];
    var board = document.getElementById('kanbanBoard');

    var filtered = allPosts.filter(function(p) {
        return p.title.toLowerCase().indexOf(query) !== -1 ||
               (p.desc && p.desc.toLowerCase().indexOf(query) !== -1) ||
               p.platform.toLowerCase().indexOf(query) !== -1 ||
               (p.designer && p.designer.toLowerCase().indexOf(query) !== -1);
    });

    board.innerHTML = statuses.map(function(status) {
        var statusPosts = filtered.filter(function(p) { return p.status === status; });
        return '<div class="kanban-column"><div class="kanban-header"><span>' + statusLabels[status] + '</span><span class="kanban-count">' + statusPosts.length + '</span></div>' + (statusPosts.length === 0 ? '<div class="empty-state"><i class="fas fa-inbox"></i><p>لا توجد نتائج</p></div>' : statusPosts.map(function(p, i) { return createCardHTML(p, i); }).join('')) + '</div>';
    }).join('');
}
